97 public Plage TrouverSousChaine(string SousChaine, int IndexDeDemarrage = 0) {
98
99 int LongueurDeLaSousChaine = SousChaine.Length;
100
101 if (LongueurDeLaSousChaine <= 0 || Compter < LongueurDeLaSousChaine) {
102
103 return new Plage();
104 }
105
106 IndexDeDemarrage += IndicesDesTampons.Demarrage;
107
108 // Rechercher le premier caractère de la sous-chaîne
109 for(int PremierCaractereIdx = IndexDeDemarrage; PremierCaractereIdx <= IndicesDesTampons.Fin; ++PremierCaractereIdx) {
110
111 if (Tampon[index: PremierCaractereIdx] != SousChaine[index: 0])
112 continue;
113
114 // Échec maintenant si la sous-chaîne ne peut pas tenir compte de la taille des et de l'index de début de recherche
115 if (PremierCaractereIdx + LongueurDeLaSousChaine - 1 > IndicesDesTampons.Fin)
116 return new Plage();
117
118 bool EstLaNonConcordanceDesChaines = false;
119 // Vérifier si la Sous-Chaîne correspond au Démarrage à l'index
120
121 for(int currentIdx = 0; currentIdx < LongueurDeLaSousChaine; ++currentIdx) {
122
123 if (Tampon[index: PremierCaractereIdx + currentIdx] != SousChaine[index: currentIdx]) {
124
125 EstLaNonConcordanceDesChaines = true;
126 break;
127 }
128 }
129
130 if (EstLaNonConcordanceDesChaines)
131 continue;
132
133 return Plage.DeL_IndexA_LaTaille(Demarrage: PremierCaractereIdx - IndicesDesTampons.Demarrage, Taille: LongueurDeLaSousChaine);
134 }
135 return new Plage();
136 }
