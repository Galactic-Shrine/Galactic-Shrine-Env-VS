153
154 if(string.IsNullOrWhiteSpace(value: Message))

157 if(string.IsNullOrWhiteSpace(value: CheminComplet))
158 throw new InvalidOperationException(message: Resources.CheminOuNomDuJournalPasDefini);
159
160 string NiveauTexte = TypePersonnalise ?? Niveau.ToString();
161
162 string Datage = $"[{DateTime.Now.ToString(format: _Horodatage, provider: CultureInfo.InvariantCulture)}] ";
163
164 string Echelon;
165
166 if(Niveau == Journalisations.Autre) {
167
168 Echelon = NiveauTexte == "Autre"
169 ? ""
170 : $"[{
171 NiveauTexte
172 .Replace(oldValue: "Info", newValue: Resources.INFO)
173 .Replace(oldValue: "Succes", newValue: Resources.SUCCES)
174 .Replace(oldValue: "Erreur", newValue: Resources.ERREUR)
175 .Replace(oldValue: "Avertissement", newValue: Resources.AVERTISSEMENT)
176 }] ";
177 }
178 else {
179
180 Echelon = $"[{NiveauTexte}] ";
181 }
182
183 string journalMessage = SousNiveau == null
184 ? $"{Datage}{Echelon}{Message}{Environment.NewLine}"
185 : $"{Datage}{Echelon}[{SousNiveau}] {Message}{Environment.NewLine}";
186
187 if(!VerifieSiExiste(Localisation: new FichierReference(Chemins: CheminComplet))) {
188
189 EcrireToutesLeslignes(Localisation: new FichierReference(Chemins: CheminComplet), Contenu: [journalMessage]);
